The hour is given as a string. Considered we have data for almost ten years, we can change the format of the hour, starting with a given year, for instance the year 2000.

import datetime as dt

start_date = dt.datetime(2000, 1, 1, int(data['Hour'][0][:2]))

data['Date'] = [start_date + dt.timedelta(hours=i) for i in range(len(data['Hour']))]

International system units#

data['Hour'] = data['Hour'].apply(lambda x: int(x.split(':')[0]))

data['Relative humidity max'] = data['Relative humidity max (%)'].apply(lambda x: x/100)
data.drop(columns=['Relative humidity max (%)'], inplace=True)

data['Relative humidity min'] = data['Relative humidity min (%)'].apply(lambda x: x/100)
data.drop(columns=['Relative humidity min (%)'], inplace=True)

data['Temperature max (K)'] = data['Temperature max (℃)'].apply(lambda x: x+273.15)
data.drop(columns=['Temperature max (℃)'], inplace=True)

data['Temperature min (K)'] = data['Temperature min (℃)'].apply(lambda x: x+273.15)
data.drop(columns=['Temperature min (℃)'], inplace=True)

data['Relative humidity'] = data['Relative humidity (%)'].apply(lambda x: x/100)
data.drop(columns=['Relative humidity (%)'], inplace=True)

data['Atmospheric pressure at station level (Pa)'] = data['Atmospheric pressure at station level (mB)'].apply(lambda x: x*100)
data.drop(columns=['Atmospheric pressure at station level (mB)'], inplace=True)

data['Precipitation (m)'] = data['Precipitation (mm)'].apply(lambda x: x/1000)
data.drop(columns=['Precipitation (mm)'], inplace=True)

data['Atmospheric pressure max (Pa)'] = data['Atmospheric pressure max (mB)'].apply(lambda x: x*100)
data.drop(columns=['Atmospheric pressure max (mB)'], inplace=True)

data['Atmospheric pressure min (Pa)'] = data['Atmospheric pressure min (mB)'].apply(lambda x: x*100)
data.drop(columns=['Atmospheric pressure min (mB)'], inplace=True)
